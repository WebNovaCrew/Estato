# Git Push Commands - Copy and Paste These

# Step 1: Navigate to project
cd C:\Estato

# Step 2: Verify Git is installed
git --version

# Step 3: Configure Git (first time only - replace with your info)
git config --global user.name "Your Name"
git config --global user.email "your.email@example.com"

# Step 4: Initialize repository
git init

# Step 5: Check status (verify .env files are NOT there)
git status

# Step 6: Add all files
git add .

# Step 7: Check status again (make sure .env is NOT included)
git status

# Step 8: Commit
git commit -m "Initial commit: Estato Real Estate App with Supabase backend and Render deployment"

# Step 9: Rename branch to main
git branch -M main

# Step 10: Add remote repository
git remote add origin https://github.com/WebNovaCrew/Estato.git

# Step 11: Push to GitHub
git push -u origin main

# If you get authentication error, use Personal Access Token:
# 1. Go to: https://github.com/settings/tokens
# 2. Generate new token (classic)
# 3. Select 'repo' scope
# 4. Copy token
# 5. When prompted for password, paste the token
